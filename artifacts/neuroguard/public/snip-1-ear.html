// Eye Aspect Ratio (EAR) — Soukupova & Cech, 2016
  // 6 landmarks per eye, indexed from MediaPipe's 468-point Face Mesh

  export const LEFT_EYE  = [362, 385, 387, 263, 373, 380];
  export const RIGHT_EYE = [ 33, 160, 158, 133, 153, 144];

  const euclideanDistance = (p1: Point, p2: Point): number => {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
  };

  export const calculateEAR = (landmarks: any[], eyeIndices: number[]): number => {
    if (!landmarks || landmarks.length === 0) return 0;

    // p1..p6 correspond to eyeIndices[0..5]
    const p1 = landmarks[eyeIndices[0]];   // outer corner
    const p2 = landmarks[eyeIndices[1]];   // upper-left
    const p3 = landmarks[eyeIndices[2]];   // upper-right
    const p4 = landmarks[eyeIndices[3]];   // inner corner
    const p5 = landmarks[eyeIndices[4]];   // lower-right
    const p6 = landmarks[eyeIndices[5]];   // lower-left

    const vertical1   = euclideanDistance(p2, p6);
    const vertical2   = euclideanDistance(p3, p5);
    const horizontal  = euclideanDistance(p1, p4);

    if (horizontal === 0) return 0;

    //          ‖p2 − p6‖ + ‖p3 − p5‖
    //  EAR  =  ─────────────────────
    //               2 · ‖p1 − p4‖
    return (vertical1 + vertical2) / (2.0 * horizontal);
  };

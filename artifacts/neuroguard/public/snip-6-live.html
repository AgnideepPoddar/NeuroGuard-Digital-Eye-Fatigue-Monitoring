export default function LiveMonitor() {
    const [selectedModel, setSelectedModel] = useState<ModelType>("cnn_lstm");
    const [activeSessionId, setActiveSessionId] = useState<number | null>(null);

    // Real-time face-mesh + EAR/PERCLOS/blink computation hook
    const {
      videoRef, canvasRef, isReady,
      currentEar, currentPerclos, blinkRate,
      fatigueState, detectionConfidence, history, modelConfig,
    } = useFaceMesh(!!activeSessionId, selectedModel);

    const { mutateAsync: createSession } = useCreateSession();
    const { mutateAsync: endSession }    = useEndSession();
    const { mutateAsync: recordMetric }  = useRecordMetric();
    const { mutateAsync: createAlert }   = useCreateAlert();

    // ── Persist a metric snapshot every 2 seconds ────────────────
    useEffect(() => {
      if (!activeSessionId || !isReady) return;
      const id = setInterval(() => {
        recordMetric({
          sessionId: activeSessionId,
          data: { ear: currentEar, perclos: currentPerclos, blinkRate, fatigueState }
        });
      }, 2000);
      return () => clearInterval(id);
    }, [activeSessionId, isReady, currentEar, currentPerclos, blinkRate, fatigueState]);

    // ── Auto-fire alert on drowsy / fatigued state ────────────────
    useEffect(() => {
      if (!activeSessionId || !isReady) return;
      if (fatigueState === 'alert') return;

      createAlert({
        sessionId: activeSessionId,
        data: {
          alertType: fatigueState,
          message: `[${modelConfig.label}] ${fatigueState.toUpperCase()} detected. ` +
                   `EAR: ${currentEar.toFixed(2)}, PERCLOS: ${(currentPerclos*100).toFixed(1)}%`,
          ear: currentEar,
          perclos: currentPerclos,
        }
      });
    }, [fatigueState, activeSessionId, isReady]);

    /* ... renders video + canvas overlay + 4 real-time charts ... */
  }
